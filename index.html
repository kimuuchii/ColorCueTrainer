    <script>

        const COLORS = [

            { name: 'แดง', nameEng: 'red', hex: '#FF3B30' },

            { name: 'เหลือง', nameEng: 'yellow', hex: '#FFCC00' },

            { name: 'เขียว', nameEng: 'green', hex: '#34C759' },

            { name: 'น้ำเงิน', nameEng: 'blue', hex: '#007AFF' },

            { name: 'ส้ม', nameEng: 'orange', hex: '#FF9500' },

            { name: 'ม่วง', nameEng: 'purple', hex: '#AF52DE' }

        ];


        let selectedColors = [];

        let gameInterval = null;

        let currentTime = 0;

        let settings = {

            colors: [0, 1, 2, 3],

            time: 5,

            voice: true,

            random: true,

            tap: true

        };


        function initColorGrid() {

            const grid = document.getElementById('colorGrid');

            COLORS.forEach((color, index) => {

                const item = document.createElement('div');

                item.className = 'color-item';

                if (settings.colors.includes(index)) {

                    item.classList.add('selected');

                }

                item.innerHTML = `

                    <div class="color-swatch" style="background-color: ${color.hex}"></div>

                    <div class="color-name">${color.name}</div>

                `;

                item.onclick = () => toggleColor(index, item);

                grid.appendChild(item);

            });

        }


        function toggleColor(index, element) {

            const colorIndex = settings.colors.indexOf(index);

            if (colorIndex > -1) {

                settings.colors.splice(colorIndex, 1);

                element.classList.remove('selected');

            } else {

                if (settings.colors.length < 6) {

                    settings.colors.push(index);

                    element.classList.add('selected');

                }

            }

            document.getElementById('colorError').classList.remove('show');

        }


        function previewColor() {

            if (settings.colors.length < 2) {

                document.getElementById('colorError').classList.add('show');

                return;

            }

            const randomIndex = settings.colors[Math.floor(Math.random() * settings.colors.length)];

            const color = COLORS[randomIndex];

            

            const playScreen = document.getElementById('playScreen');

            playScreen.style.backgroundColor = color.hex;

            document.getElementById('colorLabel').textContent = color.name;

            document.getElementById('timerDisplay').style.display = 'none';

            playScreen.classList.add('active');


            if (settings.voice) {

                speak(color.name);

            }


            setTimeout(() => {

                playScreen.classList.remove('active');

                document.getElementById('timerDisplay').style.display = 'block';

            }, 2000);

        }


        function startGame() {

            if (settings.colors.length < 2) {

                document.getElementById('colorError').classList.add('show');

                return;

            }


            settings.time = parseInt(document.getElementById('timeInput').value) || 5;

            settings.voice = document.getElementById('voiceCheck').checked;

            settings.random = document.getElementById('randomCheck').checked;

            settings.tap = document.getElementById('tapCheck').checked;


            document.getElementById('settingsScreen').style.display = 'none';

            document.getElementById('playScreen').classList.add('active');


            startCountdown();

        }


        function startCountdown() {

            let count = 3;

            const playScreen = document.getElementById('playScreen');

            const label = document.getElementById('colorLabel');

            

            playScreen.style.backgroundColor = '#333';

            label.className = 'countdown-screen';

            label.textContent = count;

            document.getElementById('timerDisplay').style.display = 'none';


            const countInterval = setInterval(() => {

                count--;

                if (count > 0) {

                    label.textContent = count;

                } else {

                    clearInterval(countInterval);

                    label.className = 'color-label';

                    document.getElementById('timerDisplay').style.display = 'block';

                    startColorGame();

                }

            }, 1000);

        }


        function startColorGame() {

            selectedColors = settings.colors.map(i => COLORS[i]);

            if (!settings.random) {

                selectedColors.sort((a, b) => COLORS.indexOf(a) - COLORS.indexOf(b));

            }

            

            let currentIndex = 0;

            let lastColorIndex = -1;

            let repeatCount = 0;

            

            function showNextColor() {

                if (settings.random) {

                    let newIndex;

                    

                    // ถ้าสีเดิมซ้ำไปแล้ว 2 ครั้ง ต้องเปลี่ยนสีใหม่

                    if (repeatCount >= 2) {

                        do {

                            newIndex = Math.floor(Math.random() * selectedColors.length);

                        } while (newIndex === lastColorIndex && selectedColors.length > 1);

                        repeatCount = 0;

                    } else {

                        newIndex = Math.floor(Math.random() * selectedColors.length);

                        

                        // ถ้าได้สีเดิม เพิ่มตัวนับ

                        if (newIndex === lastColorIndex) {

                            repeatCount++;

                        } else {

                            repeatCount = 0;

                        }

                    }

                    

                    currentIndex = newIndex;

                    lastColorIndex = newIndex;

                } else {

                    currentIndex = (currentIndex + 1) % selectedColors.length;

                }

                

                const color = selectedColors[currentIndex];

                document.getElementById('playScreen').style.backgroundColor = color.hex;

                document.getElementById('colorLabel').textContent = color.name;

                

                if (settings.voice) {

                    speak(color.name);

                }

                

                currentTime = settings.time;

                updateTimer();

            }


            function updateTimer() {

                document.getElementById('timerDisplay').textContent = `${currentTime} วิ`;

            }


            showNextColor();


            gameInterval = setInterval(() => {

                currentTime--;

                updateTimer();

                

                if (currentTime <= 0) {

                    showNextColor();

                }

            }, 1000);


            if (settings.tap) {

                document.getElementById('playScreen').onclick = (e) => {

                    if (e.target.id === 'playScreen' || e.target.id === 'colorLabel') {

                        showNextColor();

                    }

                };

            }

        }


        function stopGame() {

            clearInterval(gameInterval);

            document.getElementById('playScreen').classList.remove('active');

            document.getElementById('playScreen').onclick = null;

            document.getElementById('settingsScreen').style.display = 'block';

        }


        function speak(text) {

            if ('speechSynthesis' in window) {

                const utterance = new SpeechSynthesisUtterance(text);

                utterance.lang = 'th-TH';

                utterance.rate = 0.9;

                speechSynthesis.speak(utterance);

            }

        }


        function resetSettings() {

            settings = {

                colors: [0, 1, 2, 3],

                time: 5,

                voice: true,

                random: true,

                tap: true

            };

            

            document.getElementById('timeInput').value = 5;

            document.getElementById('voiceCheck').checked = true;

            document.getElementById('randomCheck').checked = true;

            document.getElementById('tapCheck').checked = true;

            

            document.querySelectorAll('.color-item').forEach((item, index) => {

                if (settings.colors.includes(index)) {

                    item.classList.add('selected');

                } else {

                    item.classList.remove('selected');

                }

            });

        }

        };
